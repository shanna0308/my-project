N <-1000
metasample <- rnorm(N,5,2)  
bmu <- numeric(2000)  #給bmu1000個空位

for(i in 1:2000){
  rndno <- as.integer(runif(N)*N)+1 #每次抽100個號碼
  smp <- metasample[rndno] #其中1次
  bmu[i] <- mean(smp)
 
}

p <- density(bmu)
plot(p)
qq <- quantile(bmu)
qq
##       0%      25%      50%      75%     100% 
## 4.752804 4.948462 4.994164 5.039426 5.200595
